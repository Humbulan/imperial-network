Examples
React Native
import { ImperialProvider, useImperial } from 'imperial-mobile-sdk/react';

function App() {
    return (
        <ImperialProvider apiKey="YOUR_KEY">
            <OrderScreen />
        </ImperialProvider>
    );
}

function OrderScreen() {
    const { createOrder, orders } = useImperial();
    
    const handleCreateOrder = async () => {
        await createOrder({
            description: "Maize seeds",
            amount: 500
        });
    };
    
    return (
        <View>
            <Button title="Create Order" onPress={handleCreateOrder} />
            {orders.map(order => (
                <Text key={order.id}>{order.description}</Text>
            ))}
        </View>
    );
}
